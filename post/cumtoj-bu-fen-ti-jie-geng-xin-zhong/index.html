int partition(int arr[], int left, int right)  //æ‰¾åŸºå‡†æ•° åˆ’åˆ†
{
    int i = left + 1 ;
    int j = right;
    int temp = arr[left];
 
    while(i <= j)
    {
        while (arr[i] < temp)
        {
            i++;
        }
        while (arr[j] > temp )
        {
            j--;
        }
        if (i < j)

   }

}
